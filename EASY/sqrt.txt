class Solution {
public:
    int mySqrt(int x) {
        long long int s=0,e=INT_MAX,ans=0;
        
        while(s<=e)
        {
            long long int mid=s+(e-s)/2;
            if(mid*mid<=x)
            {
                ans=mid;
                s=mid+1;
                    
            }
            else
            {
                e=mid-1;
            }
        }
        return ans;
        
    }
};
